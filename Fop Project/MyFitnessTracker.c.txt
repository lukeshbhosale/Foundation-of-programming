int SaveExerciseData(int calories_burned) {
    FILE *file = fopen("ExerciseData.txt", "a");
    if (file == NULL) {
        printf("Error opening exercise data file!\n");
        return 0;
    }
    
    time_t t = time(NULL);
    struct tm *tm_info = localtime(&t);
    char date[20];
    strftime(date, sizeof(date), "%d-%m-%Y", tm_info);
    
    fprintf(file, "Date: %s | Calories Burned: %d\n", date, calories_burned);
    fclose(file);
    printf("Exercise data saved successfully!\n");
    return 1;
}

#define MAX_FOODS 200
#define MAX_LOG 500

struct Food {
    char name[50];
    float calories;
    char unit;
};

struct LogEntry {
    char meal[20];
    char name[50];
    float quantity;
    float totalCalories;
};

struct Food foodDB[MAX_FOODS];
struct LogEntry logBook[MAX_LOG];

int foodCount = 0;
int logCount = 0;

void toLowerCase(char str[]) {
    for(int i=0; str[i]; i++)
        if(str[i]>='A' && str[i]<='Z')
            str[i]+=32;
}

int min(int a,int b,int c){
    int m=a;
    if(b<m) m=b;
    if(c<m) m=c;
    return m;
}

int editDistance(char a[], char b[]) {
    int m=strlen(a), n=strlen(b);
    int dp[50][50];

    for(int i=0;i<=m;i++) dp[i][0]=i;
    for(int j=0;j<=n;j++) dp[0][j]=j;

    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++)
            if(a[i-1]==b[j-1])
                dp[i][j]=dp[i-1][j-1];
            else
                dp[i][j]=1+min(dp[i-1][j],
                              dp[i][j-1],
                              dp[i-1][j-1]);

    return dp[m][n];
}

void loadFoods(){
    FILE *file=fopen("foods.txt","r");
    if(file==NULL){
        printf("foods.txt not found.\n");
        return;
    }

    while(fscanf(file,"%s %f %c",
        foodDB[foodCount].name,
        &foodDB[foodCount].calories,
        &foodDB[foodCount].unit)!=EOF)
        foodCount++;

    fclose(file);
}

int findFood(char name[]){

    toLowerCase(name);

    int bestIndex=-1;
    int bestDistance=100;

    for(int i=0;i<foodCount;i++){

        char temp[50];
        strcpy(temp,foodDB[i].name);
        toLowerCase(temp);

        if(strcmp(name,temp)==0)
            return i;

        int dist=editDistance(name,temp);

        if(dist<bestDistance){
            bestDistance=dist;
            bestIndex=i;
        }
    }

    if(bestDistance<=2){
        char choice;
        printf("Did you mean '%s'? (y/n): ",
               foodDB[bestIndex].name);
        scanf(" %c",&choice);

        if(choice=='y'||choice=='Y')
            return bestIndex;
        else
            return -1;
    }

    return -1;
}

void saveReport(){

    time_t t=time(NULL);
    struct tm tm=*localtime(&t);

    FILE *file=fopen("report.txt","a");
    if(file==NULL) return;

    fprintf(file,"\nDate: %02d-%02d-%04d\n",
            tm.tm_mday,
            tm.tm_mon+1,
            tm.tm_year+1900);

    float breakfast=0,lunch=0,dinner=0;

    fprintf(file,"---- Breakfast ----\n");
    for(int i=0;i<logCount;i++){
        if(strcmp(logBook[i].meal,"Breakfast")==0){
            fprintf(file,"%-12s %-10.0f %-10.2f\n",
                    logBook[i].name,
                    logBook[i].quantity,
                    logBook[i].totalCalories);
            breakfast+=logBook[i].totalCalories;
        }
    }
    fprintf(file,"Breakfast Total: %.2f kcal\n\n",breakfast);

    fprintf(file,"---- Lunch ----\n");
    for(int i=0;i<logCount;i++){
        if(strcmp(logBook[i].meal,"Lunch")==0){
            fprintf(file,"%-12s %-10.0f %-10.2f\n",
                    logBook[i].name,
                    logBook[i].quantity,
                    logBook[i].totalCalories);
            lunch+=logBook[i].totalCalories;
        }
    }
    fprintf(file,"Lunch Total: %.2f kcal\n\n",lunch);

    fprintf(file,"---- Dinner ----\n");
    for(int i=0;i<logCount;i++){
        if(strcmp(logBook[i].meal,"Dinner")==0){
            fprintf(file,"%-12s %-10.0f %-10.2f\n",
                    logBook[i].name,
                    logBook[i].quantity,
                    logBook[i].totalCalories);
            dinner+=logBook[i].totalCalories;
        }
    }
    fprintf(file,"Dinner Total: %.2f kcal\n\n",dinner);

    float totalIntake = breakfast + lunch + dinner;

    fprintf(file,"TOTAL CALORIE INTAKE: %.2f kcal\n", totalIntake);

    /* Load user info to get age for recommendation */
    LoadUserInfoFromFile();

    /* Use global variable for calories burned today */
    int burnedTotal = (int)todayCaloriesBurned;

    /* Recommended intake based on age */
    int recommended = 0;
    if(age>0){
        if(age<=3) recommended = 1000;
        else if(age<=8) recommended = 1400;
        else if(age<=13) recommended = 1800;
        else if(age<=18) recommended = 2200;
        else if(age<=30) recommended = 2400;
        else if(age<=50) recommended = 2200;
        else recommended = 2000;
    }

    fprintf(file,"Recommended Calorie Intake (age %d): %d kcal\n", age, recommended);
    fprintf(file,"Calories Burned (Exercise): %d kcal\n", burnedTotal);
    fprintf(file,"NET CALORIES (Intake - Burned): %.2f kcal\n", totalIntake - (float)burnedTotal);

    if(totalIntake - (float)burnedTotal > recommended)
        fprintf(file,"Outcome: Based on today's data you are likely to GAIN weight.\n");
    else if(totalIntake - (float)burnedTotal < recommended)
        fprintf(file,"Outcome: Based on today's data you are likely to LOSE weight.\n");
    else
        fprintf(file,"Outcome: Based on today's data your weight is likely to remain STABLE.\n");

    fclose(file);
}

void addMeal(){

    char meal[20];

    while(1){

        int mealChoice;

        printf("\nSelect Meal:\n");
        printf("1. Breakfast\n");
        printf("2. Lunch\n");
        printf("3. Dinner\n");
        printf("Enter choice: ");
        scanf("%d",&mealChoice);

        if(mealChoice==1)
            strcpy(meal,"Breakfast");
        else if(mealChoice==2)
            strcpy(meal,"Lunch");
        else if(mealChoice==3)
            strcpy(meal,"Dinner");
        else{
            printf("Invalid meal choice.\n");
            continue;
        }

        while(1){

            char foodName[50];
            printf("Enter food name: ");
            scanf("%s",foodName);

            int index=findFood(foodName);

            if(index==-1){
                printf("Food not recognized.\n");
            } else {

                float quantity;

                switch(foodDB[index].unit){
                    case 'b': printf("Enter number of bowls: "); break;
                    case 'r': printf("Enter number of rotis: "); break;
                    case 'e': printf("Enter number of eggs: "); break;
                    case 'l': printf("Enter number of plates: "); break;
                    case 'k': printf("Enter number of katoris: "); break;
                    case 'c': printf("Enter number of cups: "); break;
                    case 'm': printf("Enter quantity in ml: "); break;
                    case 'w': printf("Enter quantity in grams: "); break;
                    case 'p': printf("Enter number of fruits: "); break;
                    default: printf("Enter quantity: ");
                }

                scanf("%f",&quantity);

                float calories;

                if(foodDB[index].unit=='w' ||
                   foodDB[index].unit=='m')
                    calories=(quantity/100) *
                             foodDB[index].calories;
                else
                    calories=quantity *
                             foodDB[index].calories;

                strcpy(logBook[logCount].meal,meal);
                strcpy(logBook[logCount].name,
                       foodDB[index].name);
                logBook[logCount].quantity=quantity;
                logBook[logCount].totalCalories=calories;

                logCount++;

                printf("Added %.2f kcal.\n",calories);
            }

            int nextChoice;

            printf("\n1. Add Another Food\n");
            printf("2. Change Meal\n");
            printf("3. Main Menu\n");
            printf("Enter choice: ");
            scanf("%d",&nextChoice);

            if(nextChoice==1)
                continue;
            else if(nextChoice==2)
                break;
            else if(nextChoice==3){
                saveReport();
                return;
            }
        }
    }
}
